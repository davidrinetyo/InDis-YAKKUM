⚙️ Konfigurasi Google Sheets

Masukkan URL Google Apps Script Web App Anda untuk menyimpan data hasil assessment ke Google Sheets.

💡 Lihat panduan setup Google Sheets di bawah ini:

// Google Apps Script (Code.gs) — LENGKAP
function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var result = {status:'ok', emailSent:false};
  
  // 1. Simpan ke Sheets
  var sheet = SpreadsheetApp.openById('SHEET_ID').getActiveSheet();
  sheet.appendRow([new Date(), data.nama, data.nip,
    data.unit, data.level, data.mode,
    data.fiery, data.sunny, data.earthy, data.cool,
    data.dominant, data.secondary,
    data.elapsed_label, data.answers]);
  
  // 2. Kirim Email jika action=sendEmail
  if (data.action === 'sendEmail' && data.to) {
    GmailApp.sendEmail(data.to, data.subject, data.body,
      {cc: data.cc || '', name: 'YAKKUM Assessment'});
    result.emailSent = true;
  }
  
  return ContentService.createTextOutput(
    JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}
